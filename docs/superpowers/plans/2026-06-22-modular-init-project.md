# Modular Adaptive `init-project` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite `init-project` so it always lays a bare AI shell (CORE) and creates everything else only on explicit confirmation, backed by independent idempotent bash modules.

**Architecture:** Decompose the monolithic `scripts/scaffold-project.sh` into per-component scripts under `scripts/modules/` sharing one helper lib (`_lib.sh`). The `init-project` skill becomes an Opus-main orchestrator: it always runs `core.sh`, then in `interactive` mode asks grouped multi-select menus and runs only the selected module scripts, merging their JSON. Name and command are unchanged.

**Tech Stack:** POSIX bash (macOS BSD + Linux GNU), Markdown skill files, existing plugin assets.

## Global Constraints

- Command name stays `init-project`; this is an in-place rewrite, no rename.
- Version bump in `.claude-plugin/plugin.json` is **minor**.
- Commits in this repo carry **no AI mentions** (no `Co-Authored-By`, no "Generated by Claude") — per `skills/help/SKILL.md` rules.
- Every module is **idempotent**: never overwrite an existing file; report it in `files_skipped`. Re-run writes only what is missing.
- Every module emits exactly one JSON line: `{"status","files_written","files_skipped","warnings","error"}`. `status` ∈ `success|partial|error`; exit 0 on success/partial, 1 on error, 2 on bad args.
- Modules must be `sed`-portable (BSD + GNU) — use the `sed_inplace` helper from `_lib.sh`, never raw `sed -i ''`.
- Principle: "don't know → don't create." CORE is the only thing written without a question.
- Modules are internal: not exposed as user commands; only the orchestrator calls them.
- Test scripts live in `scripts/modules/tests/` and are runnable with plain `bash`.

```bash
# scripts/modules/tests/_assert.sh
# Minimal assertion helpers for module tests.
_PASS=0; _FAIL=0
pass() { _PASS=$((_PASS+1)); printf 'ok   - %s\n' "$1"; }
fail() { _FAIL=$((_FAIL+1)); printf 'FAIL - %s\n' "$1"; }
assert_file() { [ -f "$1" ] && pass "exists: $1" || fail "missing: $1"; }
assert_no_file() { [ ! -e "$1" ] && pass "absent: $1" || fail "should be absent: $1"; }
assert_contains() { grep -qF "$2" "$1" 2>/dev/null && pass "contains '$2': $1" || fail "missing '$2' in $1"; }
assert_json_status() { case "$1" in *"\"status\":\"$2\""*) pass "status=$2" ;; *) fail "status!=$2 in: $1" ;; esac; }
summary() { printf '\n%d passed, %d failed\n' "$_PASS" "$_FAIL"; [ "$_FAIL" -eq 0 ]; }
```

- [ ] **Step 4: Write `_lib.sh`**

```bash
# scripts/modules/_lib.sh
# Shared helpers for init-project module scripts. Source, don't execute.
set -u
WRITTEN=(); SKIPPED=(); WARNINGS=()

emit_json() {
    local status="$1" error_msg="${2:-}"
    printf '{"status":"%s","files_written":[' "$status"
    local first=1 f
    for f in "${WRITTEN[@]:-}"; do [ -z "$f" ] && continue
        [ $first -eq 1 ] || printf ','; printf '"%s"' "$f"; first=0; done
    printf '],"files_skipped":['; first=1
    for f in "${SKIPPED[@]:-}"; do [ -z "$f" ] && continue
        [ $first -eq 1 ] || printf ','; printf '"%s"' "$f"; first=0; done
    printf '],"warnings":['; first=1
    for f in "${WARNINGS[@]:-}"; do [ -z "$f" ] && continue
        [ $first -eq 1 ] || printf ','; printf '"%s"' "$f"; first=0; done
    if [ -n "$error_msg" ]; then printf '],"error":"%s"}\n' "$error_msg"
    else printf ']}\n'; fi
}
die() { WARNINGS+=("$1"); emit_json "error" "$1"; exit 1; }
mkpath() { mkdir -p "$1" || die "mkdir failed: $1"; }
write_file() {
    local path="$1" content="$2"
    if [ -e "$path" ]; then SKIPPED+=("$path"); return 0; fi
    mkpath "$(dirname "$path")"
    printf '%s' "$content" > "$path" || die "write failed: $path"
    WRITTEN+=("$path")
}
copy_asset() {
    local src="$1" dest="$2"
    [ -f "$src" ] || die "asset not found: $src"
    if [ -e "$dest" ]; then SKIPPED+=("$dest"); return 0; fi
    mkpath "$(dirname "$dest")"
    cp "$src" "$dest" || die "copy failed: $src -> $dest"
    WRITTEN+=("$dest")
}
sed_inplace() {
    local file="$1" expr="$2"
    if sed --version >/dev/null 2>&1; then sed -i "$expr" "$file"   # GNU
    else sed -i '' "$expr" "$file"; fi                              # BSD
}
PROJECT_PWD=""; PLUGIN_ROOT=""; REMAINING_ARGS=()
parse_common_args() {
    REMAINING_ARGS=()
    while [ $# -gt 0 ]; do
        case "$1" in
            --pwd) PROJECT_PWD="$2"; shift 2 ;;
            --plugin-root) PLUGIN_ROOT="$2"; shift 2 ;;
            *) REMAINING_ARGS+=("$1"); shift ;;
        esac
    done
    [ -n "$PROJECT_PWD" ] || { echo "--pwd required" >&2; exit 2; }
    [ -d "$PROJECT_PWD" ] || die "project pwd does not exist: $PROJECT_PWD"
}
```

```bash
#!/usr/bin/env bash
# agents.sh — write requested .claude/agents/*.md files.
set -u
HERE="$(cd "$(dirname "$0")" && pwd)"
. "$HERE/_lib.sh"
AGENTS=""
parse_common_args "$@"
i=0; while [ $i -lt ${#REMAINING_ARGS[@]} ]; do
    case "${REMAINING_ARGS[$i]}" in
        --agents) AGENTS="${REMAINING_ARGS[$((i+1))]}"; i=$((i+2)) ;;
        *) i=$((i+1)) ;;
    esac
done
cd "$PROJECT_PWD" || die "cannot cd to $PROJECT_PWD"

agent_body() {
    case "$1" in
        architect-reviewer) printf '%s' '---
name: architect-reviewer
description: Reviews changes against PRD, architecture docs, and active tasks
---

Read CLAUDE.md and the source-of-truth docs before reviewing any change. Verify

closes a task.
' ;;

name: backend-engineer
description: Implements backend features following the project API and DB schema
---

Read CLAUDE.md before implementing. Follow existing patterns in src/. Write or update

' ;;

name: qa-reviewer
description: Generates test scenarios and QA plans from PRD and user stories
---

Read CLAUDE.md and docs/product/prd.md. Generate scenarios covering happy path, error

' ;;

name: release-manager
description: Prepares release documentation and runs pre-release checklist
---

Read CLAUDE.md and docs/plans/tasks.md. Verify tasks complete, tests pass, docs current.
Update the changelog with the release summary.
' ;;

}
```
